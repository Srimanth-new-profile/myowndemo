let bookmarks = [{
    bookmarkId: "bookmark1",
    name: "Learning Portal",
    url: "https://learning.ccbp.in/",
}, ];

let formEl = document.getElementById("bookmarkForm");
let nameInputEl = document.getElementById("siteNameInput");
let siteUrlInputEl = document.getElementById("siteUrlInput");
let btnEL = document.getElementById("submitBtn");
let nameErrEl = document.getElementById("siteNameErrMsg");
let urlErrEl = document.getElementById("siteUrlErrMsg");


let bookmarkListEl = document.getElementById("bookmarksList");
nameInputEl.addEventListener("change", function(event) {
    if (nameInputEl.value === "") {
        nameErrEl.textContent = "Required*";
    } else {
        nameErrEl.textContent = "";
    }
});
siteUrlInputEl.addEventListener("change", function(event) {
    if (siteUrlInputEl.value === "") {
        urlErrEl.textContent = "Required*";
    } else {
        urlErrEl.textContent = "";
    }
});

function submitFormData(bookmarks) {
    let options = {
        method: "POST",
        headers: {
            "Content-Type": "application/json",
            Accept: "application/json",
            Authorization: "Bearer 94916fd03fceaa151b1a6fa3cb7340ecf3d0d9b00b8ebb257794aa9f1578ad0c",
        },
        body: JSON.stringify(bookmarks)
    };

    let urlId = bookmarks.url;

    fetch(urlId, options)
        .then(function(response) {
            return response.json();
        })
        .then(function(jsonData) {
            console.log(jsonData);
            if (jsonData.code === 422) {
                if (jsonData.data[0].message === "has already been taken") {
                    urlErrEl.textContent = "Url Already Exists";
                }
            }
        });
}

function createListItem() {
    let listItemEl = document.createElement("li");
    listItemEl.classList.add("second-container", "ml-4", "mr-4", "mb-4");
    let divContainer = document.createElement("div");
    divContainer.classList.add("d-flex", "flex-row", "justify-content-between");
    listItemEl.appendChild(divContainer);
    let secondDiv = document.createElement("div");
    secondDiv.classList.add("d-flex", "flex-column", "justify-content-center");
    divContainer.appendChild(secondDiv);
    let paraName = document.createElement("p");
    paraName.classList.add("pl-3", "pt-3", "visit-title");
    paraName.textContent = nameInputEl.value;
    secondDiv.appendChild(paraName);
    let anchorEl = document.createElement("a");

    anchorEl.href = siteUrlInputEl.value;
    anchorEl.target = "_blank";
    let buttonEl = document.createElement("button");
    buttonEl.classList.add("btn", "btn-primary", "mr-3", "mt-2", "mb-2");
    buttonEl.textContent = "Visit";
    anchorEl.appendChild(buttonEl);
    divContainer.appendChild(anchorEl);
    bookmarkListEl.appendChild(listItemEl);
}

function checkName() {
    if (nameInputEl.value === "") {
        nameErrEl.textContent = "Required*";
    } else {
        nameErrEl.textContent = "";
    }
}

function checkurl() {
    if (siteUrlInputEl.value === "") {
        urlErrEl.textContent = "Required*";
    } else {
        urlErrEl.textContent = "";
    }
}
formEl.addEventListener("submit", function(event) {
    event.preventDefault();
    checkName();
    checkurl();
    createListItem();
});
